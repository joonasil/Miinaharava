9    * Luokka menujen tapahtumien hoitamiselle.
10   * 
11   * @author Joonas
12   */
13  public class MenuHandler {
14  
15      /**
16       * Metodi hoitaa uusi peli- painikkeen toiminnan.
17       */
18      public static void handeNewGame() {
19          int x = Minesweeper.getSettings().getX();
20          int y = Minesweeper.getSettings().getY();
21          int mines = Minesweeper.getSettings().getMines();
22          Minesweeper.getTimer().reset();
23          Minefield newField = new Minefield(x, y, mines);
24          GameScreen newScreen =  new GameScreen(x, y);
25          Minesweeper.setBoard(newField);
26          Minesweeper.setScreen(newScreen);
27          Main.setScene(Minesweeper.getScreen().getScene());
28          Minesweeper.getScreen().setMinesLeft(Integer.toString(Minesweeper.getBoard().getMinesLeft()));
29          
30      }
31      
32      /**
33       * Metodi avaa huipputulos ikkunan.
34       * 
35       * @param difficulty Vaikeusaste, jonka huipputulokset avataan.
36       */
37      public static void handleHighScore(int difficulty) {
38          Minesweeper.getHighScore().load(difficulty);
39          Minesweeper.getHighScore().draw();
40      }
41  
42  }
