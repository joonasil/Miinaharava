22  
23  public class HighScoreGui {
24      Stage window;
25      HighScores hs;
26      TableView table;
27      ObservableList<Score> scores;
28      
29      public HighScoreGui() {    
30          window = new Stage();
31          window.initModality(Modality.APPLICATION_MODAL);
32          hs = new HighScores();              
33      }
34      
35      public void add(String name, String time, int difficulty) {
36          hs.add(name, time, difficulty);
37          scores.add(new Score(name, time));
38          Collections.sort(scores);
39          int x = 1;
40          for(Score s : scores) {
41              s.setPosition(x);
42              x++;
43          }
44          load(difficulty);
45      }
46      
47      public void load(int difficulty) {
48          switch (difficulty) {
49              case 0:
50                  window.setTitle("Highscores - Easy");
51                  break;
52              case 1:
53                  window.setTitle("Highscores - Medium");
54                  break;
55              case 2:
56                  window.setTitle("Highscores - Hard");
57                  break;                
58          }
59          List<Score> load = new ArrayList();
60          load = hs.load(difficulty);
61          ObservableList<Score> scores = FXCollections.observableArrayList();
62          for(Score s : load) {
63              scores.add(s);
64          }
65          this.scores = scores;
66          draw();
67      }
68      
69      public void draw() {
70          TableColumn<Score, String> posColumn = new TableColumn<>("Position");
71          posColumn.setMinWidth(80);
72          posColumn.setMaxWidth(80);
73          posColumn.setCellValueFactory(new PropertyValueFactory<>("position"));
74          TableColumn<Score, TextField> nameColumn = new TableColumn("Name");
75          nameColumn.setMinWidth(200);
76          nameColumn.setMaxWidth(200);
77          nameColumn.setCellValueFactory(new PropertyValueFactory<>("name"));
78          TableColumn<Score, String> timeColumn = new TableColumn<>("Time");
79          timeColumn.setMinWidth(90);
80          timeColumn.setMaxWidth(90);
81          timeColumn.setCellValueFactory(new PropertyValueFactory<>("time"));
82          
83          table = new TableView();
84          table.setItems(scores);
85          table.getColumns().addAll(posColumn, nameColumn, timeColumn);
86          VBox layout = new VBox();
87          HBox buttons = new HBox();
88          
89          Button easyButton = new Button("Easy"); 
90          Button mediumButton = new Button("Medium");   
91          Button hardButton = new Button("Hard"); 
92          Button closeButton = new Button("Close");
93          
94          easyButton.setOnAction(e -> load(0));
95          mediumButton.setOnAction(e -> load(1));
96          hardButton.setOnAction(e -> load(2));
97          closeButton.setOnAction(e -> window.close());
98  
99          easyButton.setMinWidth(77);
100         mediumButton.setMinWidth(77); 
101         hardButton.setMinWidth(77); 
102         closeButton.setMinWidth(77); 
103         
104         buttons.getChildren().addAll(easyButton, mediumButton, hardButton, closeButton);
105         buttons.setSpacing(18);
106         buttons.setPadding(new Insets(5,5,5,5));
107         layout.getChildren().addAll(table,buttons);
108         Scene scene = new Scene(layout, 371, 300);
109         
110         window.setScene(scene);
111         window.setResizable(false);
112         window.show();
113     }
114 }
