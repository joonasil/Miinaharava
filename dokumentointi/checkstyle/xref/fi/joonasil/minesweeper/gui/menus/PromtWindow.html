1   package fi.joonasil.minesweeper.gui.menus;
2   import javafx.event.ActionEvent;
3   import javafx.stage.*;
4   import javafx.scene.*;
5   import javafx.scene.layout.*;
6   import javafx.scene.control.*;
7   import javafx.geometry.*;
8   
9   /**
10   * Luokka luo ponnahdusikkunan, jolla voi vahvistaa käyttäjän komentoja.
11   * 
12   * @author Joonas Ilvonen
13   */
14  public class PromtWindow {
15      private static final TextField input = new TextField();
16      private static String name;
17      /**
18       * Metodi luo ponnahdusikkunan ja palauttaa käyttäjän valitseman nimimerkin.
19       * 
20       * @param title Ponnahdusikkunan otsikko
21       * @param message Ponnahdusikkunan viesti
22       * 
23       * @return käyttäjän valinta 
24       */
25      public static String gameWon(String title, String message) {
26          
27          Stage window = new Stage();
28          window.initModality(Modality.APPLICATION_MODAL);
29          window.setOnCloseRequest(e -> e.consume());
30          window.setTitle(title);
31          window.setMinWidth(275);
32          Label label = new Label();
33          label.setText(message);
34          input.setPromptText("Name");
35          Button confirm = new Button("Confirm");
36  
37          confirm.setOnAction((ActionEvent e) -> {
38              name = input.getText();
39              if(name.length() > 0) {
40                  window.close();
41              }     
42          });
43  
44          VBox layout = new VBox(10);
45  
46          layout.getChildren().addAll(label, input, confirm);
47          layout.setAlignment(Pos.CENTER);
48          Scene scene = new Scene(layout);
49          window.setScene(scene);
50          window.showAndWait();
51          return name;
52      }
53  }
