1   package fi.joonasil.minesweeper.gui.menus;
2   import javafx.stage.*;
3   import javafx.scene.*;
4   import javafx.scene.layout.*;
5   import javafx.scene.control.*;
6   import javafx.geometry.*;
7   
8   /**
9    * Luokka luo ponnahdusikkunan, jolla voi vahvistaa käyttäjän komentoja.
10   * 
11   * @author Joonas Ilvonen
12   */
13  public class PromtWindow {
14      private static boolean answer;
15      
16      /**
17       * Metodi luo ponnahdusikkunan ja palauttaa käyttäjän valitseman vastauksen.
18       * 
19       * @param title Ponnahdusikkunan otsikko
20       * @param message Ponnahdusikkunan viesti
21       * 
22       * @return käyttäjän valinta totuusarvona
23       */
24      public static boolean display(String title, String message) {
25          Stage window = new Stage();
26          window.initModality(Modality.APPLICATION_MODAL);
27          window.setTitle(title);
28          window.setMinWidth(250);
29          Label label = new Label();
30          label.setText(message);
31  
32          Button yesButton = new Button("Yes");
33          Button noButton = new Button("No");
34  
35          yesButton.setOnAction(e -> {
36              answer = true;
37              window.close();
38          });
39          noButton.setOnAction(e -> {
40              answer = false;
41              window.close();
42          });
43  
44          VBox layout = new VBox(10);
45  
46          layout.getChildren().addAll(label, yesButton, noButton);
47          layout.setAlignment(Pos.CENTER);
48          Scene scene = new Scene(layout);
49          window.setScene(scene);
50          window.showAndWait();
51  
52          return answer;
53      }
54  }
