50          
51          //Game menu items
52          RadioMenuItem easy = new RadioMenuItem("Easy");
53          RadioMenuItem medium = new RadioMenuItem("Medium");
54          RadioMenuItem hard = new RadioMenuItem("Hard");
55          
56          //Toggle group for difficulty
57          ToggleGroup dif = new ToggleGroup();
58          easy.setToggleGroup(dif);
59          medium.setToggleGroup(dif);
60          hard.setToggleGroup(dif);
61          switch(Minesweeper.getSettings().getDifficulty()){
62              case 0:
63                  dif.selectToggle(easy);
64                  break;
65              case 1:
66                  dif.selectToggle(medium);
67                  break;
68              case 2:
69                  dif.selectToggle(hard);
70                  break;
71              default:
72                  dif.selectToggle(easy);
73          }
74          
75          //Difficulty menu actions
76          easy.setOnAction(e -> Minesweeper.getSettings().setEasy());
77          medium.setOnAction(e -> Minesweeper.getSettings().setMedium());
78          hard.setOnAction(e -> Minesweeper.getSettings().setHard());
79          
80          diffMenu.getItems().addAll(easy,medium,hard);
81          
82          menuBar.getMenus().addAll(gameMenu, diffMenu);
83          return menuBar;
84      }
