22      public static MenuBar setMenus() {
23  
24          MenuBar menuBar = new MenuBar();
25          
26          Menu gameMenu = new Menu("Game");
27          
28          MenuItem newGame = new MenuItem("New Game");
29          MenuItem quitGame = new MenuItem("Quit Game");
30          
31          newGame.setOnAction(e -> MenuHandler.handeNewGame());
32          quitGame.setOnAction(e -> Main.quitGame());
33          
34          gameMenu.getItems().addAll(newGame, quitGame);
35         
36          Menu diffMenu = new Menu("Difficulty");
37          
38          RadioMenuItem easy = new RadioMenuItem("Easy");
39          RadioMenuItem medium = new RadioMenuItem("Medium");
40          RadioMenuItem hard = new RadioMenuItem("Hard");
41          
42          ToggleGroup dif = new ToggleGroup();
43          easy.setToggleGroup(dif);
44          medium.setToggleGroup(dif);
45          hard.setToggleGroup(dif);
46          switch(Minesweeper.getSettings().getDifficulty()){
47              case 0:
48                  dif.selectToggle(easy);
49                  break;
50              case 1:
51                  dif.selectToggle(medium);
52                  break;
53              case 2:
54                  dif.selectToggle(hard);
55                  break;
56              default:
57                  dif.selectToggle(easy);
58          }
59          
60          easy.setOnAction(e -> Minesweeper.getSettings().setEasy());
61          medium.setOnAction(e -> Minesweeper.getSettings().setMedium());
62          hard.setOnAction(e -> Minesweeper.getSettings().setHard());
63          
64          diffMenu.getItems().addAll(easy,medium,hard);
65          
66          Menu hsMenu = new Menu("High Scores");
67          
68          MenuItem easyScore = new MenuItem("Easy");
69          MenuItem mediumScore = new MenuItem("Medium");
70          MenuItem hardScore = new MenuItem("Hard");
71          
72          easyScore.setOnAction(e -> MenuHandler.handleHighScore(0));
73          mediumScore.setOnAction(e -> MenuHandler.handleHighScore(1));
74          hardScore.setOnAction(e -> MenuHandler.handleHighScore(2));
75          
76          hsMenu.getItems().addAll(easyScore, mediumScore, hardScore);
77          
78          menuBar.getMenus().addAll(gameMenu, diffMenu, hsMenu);
79          return menuBar;
80      }
81  }
